# DIP-Q1
import cv2
import numpy as np

h, w = 600, 600
img = np.full((h, w, 3), 255, dtype=np.uint8)

cv2.circle(img, (300, 300), 280, (0, 0, 0), 3)
cv2.circle(img, (300, 300), 5, (0, 0, 0), -1)

for i in range(12):
    angle = i * 30
    x1 = int(300 + 220 * np.cos(np.radians(angle)))
    y1 = int(300 + 220 * np.sin(np.radians(angle)))
    x2 = int(300 + 260 * np.cos(np.radians(angle)))
    y2 = int(300 + 260 * np.sin(np.radians(angle)))
    cv2.line(img, (x1, y1), (x2, y2), (0, 0, 0), 3)

cv2.line(img, (300, 300), (300, 120), (0, 0, 0), 6)
cv2.line(img, (300, 300), (170, 300), (0, 0, 0), 8)

SR_core = 1
SR_mid = 3
SR_outer1 = 6
SR_outer2 = 13

cx, cy = w // 2, h // 2
radius = min(cx, cy)
r1 = int(radius * 0.35)
r2 = int(radius * 0.70)

Y, X = np.ogrid[:h, :w]
dist = np.sqrt((X - cx)**2 + (Y - cy)**2)

mask_core = dist <= r1
mask_mid = (dist > r1) & (dist <= r2)
mask_outer = dist > r2

def sample_region(region, sr):
    sampled = region[::sr, ::sr]
    return cv2.resize(sampled, (region.shape[1], region.shape[0]), interpolation=cv2.INTER_CUBIC)

result = np.zeros_like(img)

core_resampled = sample_region(img, SR_core)
mid_resampled = sample_region(img, SR_mid)
outer_resampled1 = sample_region(img, SR_outer1)
outer_resampled2 = sample_region(img, SR_outer2)

outer_blended = cv2.addWeighted(outer_resampled1, 0.6, outer_resampled2, 0.4, 0)

result[mask_core] = core_resampled[mask_core]
result[mask_mid] = mid_resampled[mask_mid]
result[mask_outer] = outer_blended[mask_outer]

cv2.imshow("Original Synthetic Clock", img)
cv2.imshow("Non-Uniform Spatial Quantization", result)
cv2.waitKey(0)
cv2.destroyAllWindows()
